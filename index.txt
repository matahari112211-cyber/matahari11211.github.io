function showcart()
{
  $.ajax({
      url: url_common+'/showcart',
      type: 'post',
      success: function(data) {
           if(data.status == 'success'){
            $('#show_cart').html(data.data);
            $('#total_cart').html(data.total);
            $('.badge-notifications').html(data.count);
            $('[data-bs-toggle="offcanvas"][data-bs-target="#sendInvoiceOffcanvas"]').click();
           }
      }
  });
}

function deletecart(id)
{
  checked = $("input[type=checkbox]:checked").length;

  $.ajax({
      url: url_common+'/deletecart',
      type: 'post',
      data: {
        id : id
      },
      success: function(data) {
           if(data.status == 'success'){
            $('#show_cart').html(data.data);
            $('#total_cart').html(data.total);
            $('.badge-notifications').html(data.count);
            // $('[data-bs-toggle="offcanvas"][data-bs-target="#sendInvoiceOffcanvas"]').click();

             // Update stock in quantity inputs
            if (data.stock) {
              Object.keys(data.stock).forEach(function(key) {
                  let stockData = data.stock[key];
                  $('#quantity' + key).val(1);
                  $('#quantity' + key).attr('max', stockData.stock);
                  $('#quantity_desc' + key).html("Stok: <b>"+stockData.stock+"</b>");
              });
            }
           }
      }
  });
}
